## Overview

The functions module provides high-level utilities for common operations like partition switching, GPT parsing, flashing binaries, and system checks.

## System Utilities

### check_modemmanager()

Checks if ModemManager is running and exits if found.

ModemManager can interfere with serial communication to MediaTek devices. This function scans running processes and terminates the script if ModemManager is detected.

**Exits:** Process with code 1 if ModemManager is running.

```python
from functions import check_modemmanager

if __name__ == "__main__":
    check_modemmanager()
    # Safe to proceed with device communication
```

### UserInputThread

Threaded input handler for non-blocking user prompts.

```python
class UserInputThread(threading.Thread):
    def __init__(self, msg="* * * Press Enter to continue * * *", *args, **kwargs):
        ...
```

<ParamField path="msg" type="string" optional>
  Message to display to user. Defaults to "Press Enter to continue" message.
</ParamField>

**Attributes:**
- `done` (bool) - Set to True when user presses Enter
- `msg` (string) - Message displayed to user

```python
from functions import UserInputThread
import time

thread = UserInputThread(
    msg="* * * Remove hardware short now * * *\n* * * Press Enter to continue * * *"
)
thread.start()

while not thread.done:
    dev.kick_watchdog()
    time.sleep(1)

log("User confirmed, continuing...")
```

## GPT Operations

### parse_gpt(dev)

Parses GPT (GUID Partition Table) and returns partition information.

<ParamField path="dev" type="Device" required>
  Device instance with user partition active.
</ParamField>

<ResponseField name="parts" type="dict[str, tuple[int, int]]">
  Dictionary mapping partition names to (start_block, block_count) tuples.
</ResponseField>

```python
from functions import parse_gpt, switch_user

switch_user(dev)
gpt = parse_gpt(dev)

log(f"GPT parsed: {gpt}")
# Output: {'lk': (1024, 512), 'tee1': (1536, 512), 'boot': (2048, 16384), ...}

if "boot" in gpt:
    start_block, num_blocks = gpt["boot"]
    log(f"Boot partition: block {start_block}, size {num_blocks * 512} bytes")
```

## Complete Example

From `modules/main.py:22`:

```python
import sys
import time
from common import Device
from logger import log
from load_payload import load_payload, load_pl_payload
from functions import (
    check_modemmanager,
    switch_user,
    switch_boot0,
    parse_gpt,
    flash_binary,
    force_fastboot,
    UserInputThread
)

def main(dev):
    # Load payload
    if dev.preloader:
        load_pl_payload(dev)
    else:
        load_payload(dev)

    # Verify device type
    device_type_id = dev.idme_read(b"device_type_id").rstrip(b"\x00").decode("utf-8")
    log(f"Device type: {device_type_id}")

    # Parse GPT
    log("Check GPT")
    switch_user(dev)
    gpt = parse_gpt(dev)
    log(f"gpt_parsed = {gpt}")

    if "lk" not in gpt or "tee1" not in gpt:
        raise RuntimeError("bad gpt")

    # Check RPMB
    log("Check rpmb")
    rpmb = dev.rpmb_read()
    if rpmb[0:4] != b"AMZN":
        thread = UserInputThread(
            msg="rpmb looks broken; press enter to continue or Ctrl+C to abort"
        )
        thread.start()
        while not thread.done:
            dev.kick_watchdog()
            time.sleep(1)

    # Zero out RPMB to enable downgrade
    log("Downgrade rpmb")
    dev.rpmb_write(b"\x00" * 0x100)

    # Flash bootloader components
    log("Flash tz")
    switch_user(dev)
    flash_binary(dev, "../bin/tz.img", gpt["tee1"][0], gpt["tee1"][1] * 0x200)

    log("Flash lk")
    switch_user(dev)
    flash_binary(dev, "../bin/lk.bin", gpt["lk"][0], gpt["lk"][1] * 0x200)

    log("Flash lk-payload")
    switch_boot0(dev)
    flash_binary(dev, "../lk-payload/build/payload.bin", 1024)

    log("Flash preloader")
    switch_boot0(dev)
    flash_binary(dev, "../bin/preloader.img", 0)

    # Set fastboot mode and reboot
    log("Force fastboot")
    force_fastboot(dev, gpt)

    time.sleep(5)  # Wait for data to flush

    log("Reboot")
    dev.reboot()

if __name__ == "__main__":
    check_modemmanager()

    dev = Device()
    dev.find_device()

    main(dev)
```

### GPT Parsing Algorithm

From `modules/functions.py:117`:

```python
def parse_gpt(dev):
    # Read GPT header and partition entries
    data = (dev.emmc_read(0x400 // 0x200) + 
            dev.emmc_read(0x600 // 0x200) + 
            dev.emmc_read(0x800 // 0x200) + 
            dev.emmc_read(0xA00 // 0x200))
    
    num = len(data) // 0x80
    parts = dict()
    
    for x in range(num):
        part = data[x * 0x80:(x + 1) * 0x80]
        part_name = part[0x38:].decode("utf-16le").rstrip("\x00")
        part_start = struct.unpack("<Q", part[0x20:0x28])[0]
        part_end = struct.unpack("<Q", part[0x28:0x30])[0]
        parts[part_name] = (part_start, part_end - part_start + 1)
    
    return parts
```

def flash_data(dev, data, start_block, max_size=0):
    while len(data) % 0x200 != 0:
        data += b"\x00"

    if max_size and len(data) > max_size:
        raise RuntimeError("data too big to flash")

    blocks = len(data) // 0x200
    for x in range(blocks):
        print(f"[{x + 1} / {blocks}]", end='\r')
        dev.emmc_write(start_block + x, data[x * 0x200:(x + 1) * 0x200])
        if x % 10 == 0:
            dev.kick_watchdog()
    print("")
